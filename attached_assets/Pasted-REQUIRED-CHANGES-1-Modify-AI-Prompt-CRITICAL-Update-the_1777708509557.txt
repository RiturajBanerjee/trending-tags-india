2. Implement Deterministic Heat Score

Create a function:

function computeHeatScore(trend) {
  const { headlineCount, sources, startedHoursAgo, sourceHeadlines } = trend;

  const volumeScore = Math.min(headlineCount * 8, 40);
  const sourceScore = sources.length * 5;

  let freshnessScore = 0;
  if (startedHoursAgo <= 6) freshnessScore = 20;
  else if (startedHoursAgo <= 12) freshnessScore = 15;
  else if (startedHoursAgo <= 24) freshnessScore = 10;
  else freshnessScore = 5;

  const recencyPenalty = Math.min(startedHoursAgo / 10, 10);

  // Keyword repetition boost
  const words = sourceHeadlines.join(" ").toLowerCase().split(/\s+/);
  const freq = {};
  for (const w of words) {
    if (w.length > 4) freq[w] = (freq[w] || 0) + 1;
  }
  const keywordBoost = Math.min(Math.max(...Object.values(freq), 0) * 2, 10);

  let heat =
    volumeScore +
    sourceScore +
    freshnessScore -
    recencyPenalty +
    keywordBoost;

  // Clamp between 60–100
  return Math.max(60, Math.min(100, Math.round(heat)));
}
3. Implement Momentum Logic
function computeMomentum(trend) {
  const { startedHoursAgo, headlineCount } = trend;

  if (startedHoursAgo <= 6 && headlineCount >= 3) return "rising";
  if (startedHoursAgo <= 24 && headlineCount >= 4) return "peaking";
  return "cooling";
}
